001package com.minimajack.v8.transformers.impl;
002
003import com.minimajack.v8.transformers.AbstractTransformer;
004import com.minimajack.v8.utility.SerializedOutputStream;
005
006import java.io.ByteArrayOutputStream;
007import java.nio.ByteBuffer;
008import java.nio.charset.StandardCharsets;
009
010public class StringTransformer
011    implements AbstractTransformer<String>
012{
013    ThreadLocal<ByteArrayOutputStream> threadLocalValue = ThreadLocal.withInitial(() -> new ByteArrayOutputStream(256));
014
015    @Override
016    public String read(ByteBuffer buffer )
017    {
018        int value;
019        int state = 0;
020
021        ByteArrayOutputStream baos = threadLocalValue.get();
022
023        int remain = buffer.remaining();
024        while ( remain > 0 )
025        {
026            value = buffer.get();
027            remain--;
028            int prevState = state;
029            if ( value == 0x22 )
030            {
031                if ( state == 0 )
032                {
033                    state = 1;
034                }
035                else if ( state == 1 )
036                {
037                    state = 2;
038                }
039                else
040                {
041                    state = 1;
042                }
043            }
044            else if ( state == 2 )
045            {
046                buffer.position( buffer.position() - 1 );
047                break;
048            }
049            if ( prevState != 0 && state != 2 )
050            {
051                baos.write( value );
052            }
053        }
054        byte[] data = baos.toByteArray();
055        baos.reset();
056        return new String( data, StandardCharsets.UTF_8 ).intern();
057    }
058
059    @Override
060    public void write(Object object, SerializedOutputStream buffer){
061        buffer.write((byte) 0x22);
062        buffer.writeBytes(((String)object).replace("\"", "\"\"").getBytes(StandardCharsets.UTF_8));
063        buffer.write((byte) 0x22);
064    }
065
066}
