001package com.minimajack.v8.transformers.impl;
002
003import com.minimajack.v8.annotation.*;
004import com.minimajack.v8.transformers.AbstractClassTransformer;
005import com.minimajack.v8.transformers.impl.field.FieldInfo;
006import com.minimajack.v8.utility.SerializedOutputStream;
007import com.minimajack.v8.utility.V8Reader;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011import java.lang.reflect.*;
012import java.nio.ByteBuffer;
013import java.util.ArrayList;
014import java.util.HashMap;
015import java.util.List;
016import java.util.Map;
017
018public class ObjectTransformer
019    extends AbstractClassTransformer<Object>
020{
021    final Logger logger = LoggerFactory.getLogger( ObjectTransformer.class );
022
023    private static final Map<Class<?>, List<FieldInfo>> cacheFields = new HashMap<>();
024
025    private static final Map<Class<?>, Method> afterUnmarshalMethods = new HashMap<>();
026
027    private static final List<Class<?>> versionalClasses = new ArrayList<>();
028
029    @Override
030    public Object read( Class<?> clazz, ByteBuffer buffer )
031    {
032        boolean isV8Class = clazz.getAnnotation( V8Class.class ) != null;
033        Object object;
034        if ( isV8Class )
035        {
036            return readV8Class( clazz, buffer);
037        }
038        else
039        {
040            object = readClass( clazz, buffer);
041        }
042
043        return object;
044    }
045
046    @Override
047    public void write(Object object, SerializedOutputStream buffer) {
048        Class clazz = object.getClass();
049        boolean isV8Class = clazz.getAnnotation( V8Class.class ) != null;
050        if ( isV8Class )
051        {
052            writeV8Class( object, clazz, buffer );
053        }
054        else
055        {
056            writeClass( object, clazz, buffer);
057        }
058    }
059
060    private void writeClass(Object object, Class clazz, SerializedOutputStream buffer) {
061        {
062            List<FieldInfo> fields = cacheFields.get( clazz );
063            if ( fields == null )
064            {
065                fields = processClass( clazz );
066            }
067            boolean first = true;
068            boolean hasVersion = versionalClasses.contains( clazz );
069            Integer version = 0;
070            try
071            {
072                for ( FieldInfo field : fields )
073                {
074                    if ( hasVersion )
075                    {
076                        V8Since since = field.since;
077                        if ( since != null )
078                        {
079                            if ( version < since.version() || version >= since.removed() )
080                            {
081                                continue;
082                            }
083                        }
084                    }
085                    if ( !field.required  )
086                    {
087                        if(field.field.get(object) == null)
088                        break;
089                    }
090                    if ( !first )
091                    {
092                        buffer.putComa();
093                    }
094                    Class<?> fieldType = field.fieldType;
095                    logger.debug( "Write field {} {}", field.name, fieldType );
096                    Type paramType = field.paramType;
097                    V8Reader.write( field.field.get(object), paramType, buffer );
098                    if ( field.isVersion )
099                    {
100                        hasVersion = true;
101                        version = (Integer) field.field.get(object);
102                    }
103
104                    first = false;
105
106                }
107            }
108            catch ( IllegalArgumentException | IllegalAccessException e )
109            {
110                throw new RuntimeException( e );
111            }
112
113        }
114    }
115
116    public void writeV8Class( Object object, Class<?> clazz, SerializedOutputStream buffer)
117    {
118        buffer.putOpenBracket();
119
120        writeClass( object, clazz, buffer );
121
122        buffer.putCloseBracket();
123
124    }
125
126    public Object readV8Class( Class<?> clazz, ByteBuffer buffer)
127    {
128        readBracket( buffer );
129
130        Object object = readClass( clazz, buffer );
131
132        readCloseBracket( buffer );
133
134        runAfterUnmarshal( clazz, object );
135
136        return object;
137    }
138
139    public List<FieldInfo> processClass( Class<?> clazz )
140    {
141        Field[] fields = clazz.getDeclaredFields();
142
143        V8Version v8clver = clazz.getAnnotation( V8Version.class );
144        if ( v8clver != null )
145        {
146            if ( !fields[0].getName().equals( "version" ) )
147            {
148                throw new RuntimeException( "bad version: " + clazz.getName() );
149            }
150        }
151        List<FieldInfo> lfi = new ArrayList<>();
152        for ( Field field : fields )
153        {
154            if ( field.getAnnotation( V8Transient.class ) != null )
155            {
156                continue;
157            }
158            if ( ( field.getModifiers() & java.lang.reflect.Modifier.FINAL ) == java.lang.reflect.Modifier.FINAL )
159            {
160                continue;
161            }
162            if ( ( field.getModifiers() & Modifier.PRIVATE ) == java.lang.reflect.Modifier.PRIVATE )
163            {
164                field.setAccessible(true);
165            }
166
167            FieldInfo fi = new FieldInfo();
168            fi.field = field;
169            fi.fieldType = field.getType();
170            fi.paramType = field.getGenericType();
171            fi.since = field.getAnnotation( V8Since.class );
172            fi.name = field.getName();
173            Required req = field.getAnnotation( Required.class );
174            fi.required = req == null || req.required();
175            if ( field.getAnnotation( V8Version.class ) != null )
176            {
177                fi.isVersion = true;
178                versionalClasses.add( clazz );
179            }
180            lfi.add( fi );
181        }
182        cacheFields.put( clazz, lfi );
183
184        for ( Method method : clazz.getMethods() )
185        {
186            if ( method.getName().equalsIgnoreCase( "afterUnmarshal" ) )
187            {
188                afterUnmarshalMethods.put( clazz, method );
189                break;
190            }
191        }
192        return lfi;
193    }
194
195    public Object readClass( Class<?> clazz, ByteBuffer buffer )
196    {
197        Object object;
198        try
199        {
200            object = clazz.getConstructor().newInstance();
201        }
202        catch (InstantiationException | IllegalAccessException | NoSuchMethodException | InvocationTargetException e )
203        {
204            throw new RuntimeException( "Cant instantinate class" + clazz );
205        }
206
207        List<FieldInfo> fields = cacheFields.get( clazz );
208        if ( fields == null )
209        {
210            fields = processClass( clazz );
211        }
212        boolean first = true;
213        boolean hasVersion = versionalClasses.contains( clazz );
214        Integer version = 0;
215        try
216        {
217            for ( FieldInfo field : fields )
218            {
219                if ( hasVersion )
220                {
221                    V8Since since = field.since;
222                    if ( since != null )
223                    {
224                        if ( version < since.version() || version >= since.removed() )
225                        {
226                            continue;
227                        }
228                    }
229                }
230                byte nextSymbol = buffer.get(buffer.position());
231                if ( field.required || nextSymbol != 0x7D
232                    && nextSymbol != 0x0D )
233                {
234                    if ( !first )
235                    {
236                        V8Reader.readChar( buffer, ',' );
237                    }
238                }
239                else
240                {
241                    break;
242                }
243                Class<?> fieldType = field.fieldType;
244                logger.debug( "Read field {} {}", field.name, fieldType );
245                Type paramType = field.paramType;
246                Object fieldValue = V8Reader.read( fieldType, paramType, buffer );
247                if ( hasVersion && field.isVersion )
248                {
249                    version = (Integer) fieldValue;
250                }
251                logger.debug( "VALUE: {} ", fieldValue );
252                logger.debug( "EndRead: {} ", field.name );
253
254                field.field.set( object, fieldValue );
255                first = false;
256
257            }
258        }
259        catch ( IllegalArgumentException | IllegalAccessException e )
260        {
261            throw new RuntimeException( e );
262        }
263
264        return object;
265    }
266
267    private void runAfterUnmarshal( Class<?> clazz, Object object )
268    {
269        try
270        {
271            Method method = afterUnmarshalMethods.get( clazz );
272            if ( method != null )
273            {
274                method.invoke( object );
275            }
276        }
277        catch ( SecurityException | IllegalAccessException | IllegalArgumentException | InvocationTargetException e1 )
278        {
279            throw new RuntimeException( "Can't run afterUnmarshal method" );
280        }
281    }
282
283    public static void readBracket( ByteBuffer buffer )
284    {
285        V8Reader.readChar( buffer, '{' );
286    }
287
288    public static void readCloseBracket( ByteBuffer buffer )
289    {
290        V8Reader.readChar( buffer, '}' );
291    }
