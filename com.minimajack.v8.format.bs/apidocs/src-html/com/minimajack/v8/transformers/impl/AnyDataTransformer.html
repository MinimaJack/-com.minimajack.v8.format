001package com.minimajack.v8.transformers.impl;
002
003import com.minimajack.v8.transformers.AbstractTransformer;
004import com.minimajack.v8.utility.AnyData;
005import com.minimajack.v8.utility.SerializedOutputStream;
006
007import java.nio.ByteBuffer;
008import java.nio.charset.StandardCharsets;
009
010/**
011 * Read all except comma and brackets
012 * @author e.vanzhula
013 *
014 */
015public class AnyDataTransformer
016    implements AbstractTransformer<AnyData>
017{
018    /**
019     * <b>,</b> ascii - code
020     */
021    private static final byte COMMA = 0x2C;
022
023    /**
024     * <b>{</b> - ascii code 
025     */
026    private static final byte START_BKT = 0x7B;
027
028    /**
029     * <b>}</b> - ascii code 
030     */
031    private static final byte END_BKT = 0x7D;
032
033    @Override
034    public AnyData read( ByteBuffer buffer )
035    {
036        StringBuilder sb = new StringBuilder();
037
038        while ( buffer.hasRemaining() )
039        {
040            int value = buffer.get();
041            if ( value != COMMA && value != START_BKT && value != END_BKT )
042            {
043                sb.append( (char) value );
044            }
045            else
046            {
047                buffer.position( buffer.position() - 1 );
048                break;
049            }
050        }
051        return new AnyData( sb.toString() );
052    }
053
054    @Override
055    public void write(Object data, SerializedOutputStream buffer) {
056        buffer.writeBytes(((AnyData)data).getData().getBytes(StandardCharsets.UTF_8));
057    }
058
059}
