001package com.minimajack.v8.transformers.impl;
002
003import com.minimajack.v8.transformers.AbstractTransformer;
004import com.minimajack.v8.utility.SerializedOutputStream;
005
006import java.nio.ByteBuffer;
007import java.nio.charset.StandardCharsets;
008
009public class IntegerTransformer
010    implements AbstractTransformer<Integer>
011{
012
013    @Override
014    public Integer read( ByteBuffer buffer )
015    {
016
017        StringBuilder sb = new StringBuilder();
018        boolean first = true;
019        while ( buffer.hasRemaining() )
020        {
021            int value = buffer.get();
022            if ( value >= 0x30 && value <= 0x3A )
023            {
024                sb.append( (char) value );
025            }
026            else if ( first && value >= 0x2D )
027            {
028                sb.append( (char) value );
029            }
030            else
031            {
032                buffer.position( buffer.position() - 1 );
033                break;
034            }
035            first = false;
036        }
037
038        return Long.valueOf( sb.toString() ).intValue();
039    }
040
041    @Override
042    public void write(Object object, SerializedOutputStream buffer) {
043
044        buffer.writeBytes(String.valueOf(object).getBytes(StandardCharsets.UTF_8));
045    }
046
047}
