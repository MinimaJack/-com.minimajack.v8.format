001package com.minimajack.v8.transformers;
002
003import com.minimajack.v8.utility.SerializedOutputStream;
004
005import java.lang.reflect.ParameterizedType;
006import java.nio.ByteBuffer;
007
008public interface AbstractParametrizedTransformer<T> {
009    T read(ParameterizedType type, ByteBuffer buffer);
010
011    default T read(ByteBuffer buffer) {
012        return this.read(null, buffer);
013    }
014
015    void write(Object object, SerializedOutputStream buffer);
016}
