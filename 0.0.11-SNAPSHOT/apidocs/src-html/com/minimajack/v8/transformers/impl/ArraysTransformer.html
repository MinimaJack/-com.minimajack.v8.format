001package com.minimajack.v8.transformers.impl;
002
003import java.lang.reflect.Array;
004import java.nio.ByteBuffer;
005import java.util.LinkedList;
006import java.util.List;
007
008import com.minimajack.v8.transformers.AbstractArraysTransformer;
009import com.minimajack.v8.utility.V8Reader;
010
011public class ArraysTransformer
012    extends AbstractArraysTransformer<Object[]>
013{
014
015    @Override
016    public Object[] read( Class<?> clazz, ByteBuffer buffer )
017    {
018        Class<?> componentType = clazz.getComponentType();
019        List<Object> list = new LinkedList<Object>();
020        boolean readNext = true;
021        while ( readNext )
022        {
023            list.add( V8Reader.read( componentType, buffer ) );
024            readNext = buffer.hasRemaining() && buffer.get( buffer.position() ) == 0x2C;
025            if ( readNext )
026            {
027                V8Reader.readChar( buffer, ',' );
028            }
029        }
030        Object[] arr = (Object[]) Array.newInstance( componentType, list.size() );
031        System.arraycopy( list.toArray(), 0, arr, 0, list.size() );
032        return arr;
033    }
034
035}
