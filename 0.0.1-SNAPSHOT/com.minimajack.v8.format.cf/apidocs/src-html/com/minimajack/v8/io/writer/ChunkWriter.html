001package com.minimajack.v8.io.writer;
002
003import java.io.ByteArrayOutputStream;
004import java.io.DataOutput;
005import java.io.IOException;
006
007import com.google.common.io.LittleEndianDataOutputStream;
008import com.minimajack.v8.format.BlockHeader;
009import com.minimajack.v8.io.metrix.ChunkSizeResolver;
010import com.minimajack.v8.io.metrix.SizeResolver;
011import com.minimajack.v8.model.BufferedObject;
012import com.minimajack.v8.model.FakeMemoryAllocator;
013
014public abstract class ChunkWriter
015{
016
017    private final ByteArrayOutputStream outputBuffer = new ByteArrayOutputStream();
018
019    private final LittleEndianDataOutputStream dataOutputStream = new LittleEndianDataOutputStream( outputBuffer );
020
021    private final FakeMemoryAllocator allocator = new FakeMemoryAllocator();
022
023    private final BlockHeader cursor = new BlockHeader();
024
025    private ByteArrayOutputStream chunkBuffer;
026
027    private LittleEndianDataOutputStream chunkBufferOutputStream;
028
029    private ChunkSizeResolver sizeResolver;
030
031    private int position = 0;
032
033    private int limit;
034
035    public ChunkWriter()
036    {
037        super();
038    }
039
040    public void write( BufferedObject bo )
041        throws IOException
042    {
043        bo.write( chunkBufferOutputStream );
044        flushChunks();
045    }
046
047    public void write( byte[] b )
048        throws IOException
049    {
050        chunkBuffer.write( b );
051        flushChunks();
052    }
053
054    private void flushChunks()
055    {
056        position = chunkBuffer.size();
057        if ( position > limit )
058        {
059            int offset = 0;
060            byte[] data = chunkBuffer.toByteArray();
061            while ( position > limit )
062            {
063                int sizeToWrite = ( position / limit ) * limit;
064                while ( sizeToWrite > offset )
065                {
066                    offset += dataReady( data, offset, limit );
067                }
068                position -= offset;
069            }
070            chunkBuffer.reset();
071            chunkBuffer.write( data, offset, data.length - offset );
072        }
073    }
074
075    public void allocateData( int size )
076    {
077        allocator.allocate( size );
078    }
079
080    public int getPosition()
081    {
082        return allocator.getAllocationSize();
083    }
084
085    public void writeAllData()
086    {
087        this.allocateData( this.sizeResolver.getChunkSize() + BlockHeader.V8_HEADERSIZE );
088        while ( hasData() )
089        {
090            getDataToWrite();
091        }
092        flush();
093    }
094
095    public abstract void getDataToWrite();
096
097    public boolean hasData()
098    {
099        return position > limit;
100    }
101
102    public void flush()
103    {
104        byte[] data = new byte[limit];
105        byte[] tempData = chunkBuffer.toByteArray();
106        for ( int i = 0; i < tempData.length; i++ )
107        {
108            data[i] = tempData[i];
109        }
110        fillChunk( data, 0, limit );
111    }
112
113    private void fillChunk( byte[] b, int off, int len )
114    {
115        try
116        {
117            cursor.write( dataOutputStream );
118            dataOutputStream.write( b, off, len );
119            cursor.setDocSize( 0 );
120            cursor.setNextBlock( Integer.MAX_VALUE );
121            onFillBlock();
122        }
123        catch ( IOException e )
124        {
125            throw new RuntimeException( "Can't write readed data" );
126        }
127    }
128
129    protected void onFillBlock()
130    {
131
132    }
133
134    private int dataReady( byte[] b, int off, int len )
135    {
136        if ( this.hasData() )
137        {
138            this.cursor.setNextBlock( getPosition() );
139        }
140        fillChunk( b, off, len );
141        this.allocateData( sizeResolver.getChunkSize() + BlockHeader.V8_HEADERSIZE );
142        return len;
143    }
144
145    public SizeResolver getSizeResolver()
146    {
147        return sizeResolver;
148    }
149
150    public void setFullSize( int fullSize )
151    {
152        this.cursor.setDocSize( fullSize );
153    }
154
155    public void setSizeResolver( ChunkSizeResolver sizeResolver )
156    {
157        this.sizeResolver = sizeResolver;
158        this.limit = this.sizeResolver.getChunkSize();
159        this.chunkBuffer = new ByteArrayOutputStream( this.limit );
160        this.chunkBufferOutputStream = new LittleEndianDataOutputStream( chunkBuffer );
161        this.cursor.setBlockSize( this.limit );
162    }
163
164    public byte[] getRawData()
165    {
166        return outputBuffer.toByteArray();
167    }
168
169    public DataOutput getDataOutputStream()
170    {
171        return dataOutputStream;
172    }
173
174    public int getMaximumBufferSize()
175    {
176        if ( outputBuffer.size() == 0 )
177        {
178            writeAllData();
179        }
180        return outputBuffer.size();
181    }
182
183    protected void writeData()
184    {
185
186    }
187
188}
