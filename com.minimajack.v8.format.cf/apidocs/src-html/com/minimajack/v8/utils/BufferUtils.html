001package com.minimajack.v8.utils;
002
003import java.io.DataOutput;
004import java.io.IOException;
005import java.nio.ByteBuffer;
006
007public class BufferUtils
008{
009
010    public static final int getLongFromString( ByteBuffer buffer )
011        throws IOException
012    {
013        byte[] stringBuffer = new byte[8];
014        buffer.get( stringBuffer );
015        buffer.get(); // space
016        return (int) Long.parseLong( new String( stringBuffer ), 16 );
017    }
018
019    public static final void writeLongToString( DataOutput stream, long value )
020        throws IOException
021    {
022        String formatted = String.format( "%08x ", value );
023        stream.write( formatted.getBytes() );
024    }
025}
