001package com.minimajack.v8.format;
002
003import com.google.common.io.ByteStreams;
004import com.minimajack.v8.io.stream.V8InputStream;
005import com.minimajack.v8.model.BufferedObject;
006import com.minimajack.v8.utils.BufferUtils;
007
008import java.io.DataOutput;
009import java.io.IOException;
010import java.nio.ByteBuffer;
011
012/**
013 * @author e.vanzhula
014 * 
015 */
016public class BlockHeader
017    extends BufferedObject
018{
019    /**
020     * Full size for first block or zero for the next
021     */
022    private int payloadSize;
023
024    /**
025     * Block size - without header size
026     */
027    private int blockSize;
028
029    /**
030     * Next block address or Integer.MAX_VALUE if next block does not exist
031     */
032    private int nextBlockPosition = Integer.MAX_VALUE;
033
034    /**
035     * temporary flag to read header
036     * 
037     * TODO:remove unneeded operation
038     */
039    private boolean ready = false;
040
041    /**
042     * Basic v8 marker
043     */
044    public static final short V8_SEPARATOR = 0x0A0D;
045
046    /**
047     * End block marker
048     */
049    public static final int V8_ENDBLOCK = Integer.MAX_VALUE;
050
051    /**
052     * Full header size
053     */
054    public static final int V8_HEADERSIZE = 2 + 9 + 9 + 9 + 2;
055
056    public BlockHeader()
057    {
058        setReady( true );
059    }
060
061    public BlockHeader( ByteBuffer buffer )
062    {
063        super( buffer );
064    }
065
066    public BlockHeader( ByteBuffer buffer, int position )
067    {
068        super( buffer, position );
069    }
070
071    public boolean hasNext()
072        throws IOException
073    {
074        return !this.getNextBlock().equals( Integer.MAX_VALUE );
075    }
076
077    public void read()
078        throws IOException
079    {
080        readHeader();
081    }
082
083    @Override
084    public void write( DataOutput dos )
085        throws IOException
086    {
087        dos.writeShort( V8_SEPARATOR );
088        BufferUtils.writeLongToString( dos, this.payloadSize );
089        BufferUtils.writeLongToString( dos, this.blockSize );
090        BufferUtils.writeLongToString( dos, this.nextBlockPosition );
091        dos.writeShort( V8_SEPARATOR );
092    }
093
094    /**
095     * Read data-stream to byte array
096     * 
097     * @return data
098     * @throws IOException
099     *             if can't read full data or stream broken
100     */
101    public byte[] readFully()
102        throws IOException
103    {
104        byte[] data = new byte[this.getDocSize()];
105        ByteStreams.readFully( new V8InputStream( this ), data );
106        return data;
107    }
108
109    public void reset()
110        throws IOException
111    {
112        this.readHeader();
113    }
114
115    public boolean isReady()
116    {
117        return ready;
118    }
119
120    public void setReady( boolean ready )
121    {
122        this.ready = ready;
123    }
124
125    public void readHeader()
126        throws IOException
127    {
128        this.setStartPosition();
129        ByteBuffer buffer = getBuffer();
130        buffer.getShort();
131        this.payloadSize = BufferUtils.getLongFromString( buffer );
132        this.blockSize = BufferUtils.getLongFromString( buffer );
133        this.nextBlockPosition = BufferUtils.getLongFromString( buffer );
134        buffer.getShort();
135        this.ready = true;
136    }
137
138    public V8InputStream getInputStream()
139        throws IOException
140    {
141        return new V8InputStream( this );
142    }
143
144    public byte[] getRawData()
145        throws IOException
146    {
147        byte[] rawdata = new byte[this.getBlockSize()];
148        getBuffer().get( rawdata );
149        return rawdata;
150    }
151
152    public byte[] getDocData()
153        throws IOException
154    {
155        this.readHeader();
156        if ( this.getDocSize() > this.getBlockSize() )
157        {
158            throw new RuntimeException( "Document size greather then block in attributes" );
159        }
160        byte[] rawdata = new byte[this.getDocSize()];
161        getBuffer().get( rawdata );
162        return rawdata;
163    }
164
165    public int getDocSize()
166        throws IOException
167    {
168        if ( !this.ready )
169        {
170            this.readHeader();
171        }
172        return payloadSize;
173    }
174
175    public void setDocSize( int docSize )
176    {
177        this.payloadSize = docSize;
178    }
179
180    public int getBlockSize()
181        throws IOException
182    {
183        if ( !this.ready )
184        {
185            this.readHeader();
186        }
187        return blockSize;
188    }
189
190    public void setBlockSize( int blockSize )
191    {
192        this.blockSize = blockSize;
193    }
194
195    public Integer getNextBlock()
196        throws IOException
197    {
198        if ( !this.ready )
199        {
200            this.readHeader();
201        }
202        return nextBlockPosition;
203    }
204
205    public void setNextBlock( int nextBlock )
206    {
207        this.nextBlockPosition = nextBlock;
208    }
209
210    @Override
211    public int getSize()
212        throws IOException
213    {
214        throw new RuntimeException( "Can't get real size" );
215    }
216
217}
