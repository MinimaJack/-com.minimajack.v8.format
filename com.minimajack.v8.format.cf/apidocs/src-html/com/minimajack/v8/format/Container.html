001package com.minimajack.v8.format;
002
003import com.google.common.io.ByteStreams;
004import com.minimajack.v8.model.BufferedObject;
005
006import java.io.*;
007import java.nio.ByteBuffer;
008import java.nio.ByteOrder;
009import java.util.zip.Inflater;
010import java.util.zip.InflaterInputStream;
011
012/**
013 * @author e.vanzhula
014 *
015 * Basic container. Contains other data.
016 */
017public class Container
018    extends BufferedObject
019{
020
021    private int freeBlock = Integer.MAX_VALUE;
022
023    private int sizeBlock = 512;
024
025    private int version;
026
027    private int reserved;
028
029    private byte[] data;
030
031    public static final int CONTAINER_SIZE = 4 * 4;
032
033    private V8FileSystem fileSystem;
034
035    public Container()
036    {
037
038    }
039
040    public Container( ByteBuffer buffer )
041    {
042        super( buffer );
043    }
044
045    public Container( byte[] data )
046    {
047        this.data = data;
048    }
049
050    @Override
051    public void write( DataOutput buffer )
052        throws IOException
053    {
054        buffer.writeInt( this.freeBlock );
055        buffer.writeInt( this.sizeBlock );
056        buffer.writeInt( this.version );
057        buffer.writeInt( this.reserved );
058    }
059
060    public void read()
061        throws IOException
062    {
063        if ( !hasBuffer() )
064        {
065            initBuffer();
066        }
067        ByteBuffer buffer = getBuffer();
068        this.freeBlock = buffer.getInt();
069        this.sizeBlock = buffer.getInt();
070        this.version = buffer.getInt();
071        this.reserved = buffer.getInt();
072        fileSystem = new V8FileSystem( buffer );
073        fileSystem.setContext( getContext() );
074    }
075
076    public void cleanUp()
077    {
078        getBuffer().clear();
079        this.data = null;
080    }
081
082    private void initBuffer()
083        throws IOException
084    {
085
086        ByteArrayOutputStream baos = new ByteArrayOutputStream();
087        InputStream dataStream = new InflaterInputStream( new ByteArrayInputStream( data ), new Inflater( true ) );
088        ByteStreams.copy( dataStream, baos );
089        ByteBuffer bb = ByteBuffer.wrap( baos.toByteArray() );
090        bb.order( ByteOrder.LITTLE_ENDIAN );
091        setBuffer( bb );
092        this.data = null;
093    }
094
095    public long getFreeBlock()
096    {
097        return freeBlock;
098    }
099
100    public void setFreeBlock( int freeBlock )
101    {
102        this.freeBlock = freeBlock;
103    }
104
105    public int getSizeBlock()
106    {
107        return sizeBlock;
108    }
109
110    public void setSizeBlock( int sizeBlock )
111    {
112        this.sizeBlock = sizeBlock;
113    }
114
115    public int getVersion()
116    {
117        return version;
118    }
119
120    public void setVersion( int version )
121    {
122        this.version = version;
123    }
124
125    public synchronized void updateVersion()
126    {
127        this.version++;
128    }
129
130    public int getReserved()
131    {
132        return reserved;
133    }
134
135    public void setReserved( int reserved )
136    {
137        this.reserved = reserved;
138    }
139
140    public V8FileSystem getFileSystem()
141    {
142        return fileSystem;
143    }
144
145    public void setFileSystem( V8FileSystem fileSystem )
146    {
147        this.fileSystem = fileSystem;
148    }
149
150    @Override
151    public int getSize()
152        throws IOException
153    {
154        return CONTAINER_SIZE;
155    }
156}
