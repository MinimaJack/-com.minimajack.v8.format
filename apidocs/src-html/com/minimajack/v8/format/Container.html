001package com.minimajack.v8.format;
002
003import java.io.ByteArrayInputStream;
004import java.io.ByteArrayOutputStream;
005import java.io.DataOutput;
006import java.io.IOException;
007import java.io.InputStream;
008import java.nio.ByteBuffer;
009import java.nio.ByteOrder;
010import java.util.zip.Inflater;
011import java.util.zip.InflaterInputStream;
012
013import com.google.common.io.ByteStreams;
014import com.minimajack.v8.model.BufferedObject;
015
016/**
017 * @author e.vanzhula
018 *
019 * Basic container. Contains other data.
020 */
021public class Container
022    extends BufferedObject
023{
024
025    private int freeBlock = Integer.MAX_VALUE;
026
027    private int sizeBlock = 512;
028
029    private int version;
030
031    private int reserved;
032
033    private byte[] data;
034
035    public static final int CONTAINER_SIZE = 4 * 4;
036
037    private V8FileSystem fileSystem;
038
039    public Container()
040    {
041
042    }
043
044    public Container( ByteBuffer buffer )
045    {
046        super( buffer );
047    }
048
049    public Container( byte[] data )
050    {
051        this.data = data;
052    }
053
054    @Override
055    public void write( DataOutput buffer )
056        throws IOException
057    {
058        buffer.writeInt( this.freeBlock );
059        buffer.writeInt( this.sizeBlock );
060        buffer.writeInt( this.version );
061        buffer.writeInt( this.reserved );
062    }
063
064    public void read()
065        throws IOException
066    {
067        if ( !hasBuffer() )
068        {
069            initBuffer();
070        }
071        ByteBuffer buffer = getBuffer();
072        this.freeBlock = buffer.getInt();
073        this.sizeBlock = buffer.getInt();
074        this.version = buffer.getInt();
075        this.reserved = buffer.getInt();
076        fileSystem = new V8FileSystem( buffer );
077        fileSystem.setContext( getContext() );
078    }
079
080    public void cleanUp()
081    {
082        getBuffer().clear();
083        this.data = null;
084    }
085
086    private void initBuffer()
087        throws IOException
088    {
089
090        ByteArrayOutputStream baos = new ByteArrayOutputStream();
091        InputStream dataStream = new InflaterInputStream( new ByteArrayInputStream( data ), new Inflater( true ) );
092        ByteStreams.copy( dataStream, baos );
093        ByteBuffer bb = ByteBuffer.wrap( baos.toByteArray() );
094        bb.order( ByteOrder.LITTLE_ENDIAN );
095        setBuffer( bb );
096        this.data = null;
097    }
098
099    public long getFreeBlock()
100    {
101        return freeBlock;
102    }
103
104    public void setFreeBlock( int freeBlock )
105    {
106        this.freeBlock = freeBlock;
107    }
108
109    public int getSizeBlock()
110    {
111        return sizeBlock;
112    }
113
114    public void setSizeBlock( int sizeBlock )
115    {
116        this.sizeBlock = sizeBlock;
117    }
118
119    public int getVersion()
120    {
121        return version;
122    }
123
124    public void setVersion( int version )
125    {
126        this.version = version;
127    }
128
129    public synchronized void updateVersion()
130    {
131        this.version++;
132    }
133
134    public int getReserved()
135    {
136        return reserved;
137    }
138
139    public void setReserved( int reserved )
140    {
141        this.reserved = reserved;
142    }
143
144    public V8FileSystem getFileSystem()
145    {
146        return fileSystem;
147    }
148
149    public void setFileSystem( V8FileSystem fileSystem )
150    {
151        this.fileSystem = fileSystem;
152    }
153
154    @Override
155    public int getSize()
156        throws IOException
157    {
158        return CONTAINER_SIZE;
159    }
160}
