001package com.minimajack.v8.metadata.style.item.font;
002
003import com.minimajack.v8.metadata.inner.enums.FontType;
004import com.minimajack.v8.metadata.inner.enums.FontType.Values;
005import com.minimajack.v8.metadata.style.item.color.IntObject;
006import com.minimajack.v8.metadata.style.item.font.impl.AbsoluteFont;
007import com.minimajack.v8.metadata.style.item.font.impl.StyleItemFont;
008import com.minimajack.v8.metadata.style.item.font.impl.WindowsFont;
009import com.minimajack.v8.transformers.AbstractTransformer;
010import com.minimajack.v8.utility.SerializedOutputStream;
011import com.minimajack.v8.utility.V8Reader;
012
013import java.nio.ByteBuffer;
014
015public class FontTransformer implements AbstractTransformer<FontDescription> {
016
017  private static final int FONT_WEIGHT = 1 << 5;
018  private static final int FONT_UNK1 = 1 << 4;
019  private static final int FONT_UNK2 = 1 << 3;
020  private static final int FONT_UNK3 = 1 << 2;
021  private static final int FONT_UNK4 = 1 << 1;
022  private static final int FONT_UNK5 = 1;
023
024  @Override
025  public FontDescription read(final ByteBuffer buffer) {
026    FontDescription innerFont = null;
027    final Values typeFont = V8Reader.read(FontType.Values.class, buffer);
028    switch (typeFont) {
029      case Absolute:
030        V8Reader.readChar(buffer, ',');
031        innerFont = V8Reader.read(AbsoluteFont.class, buffer);
032      break;
033      case StyleItem:
034        final StyleItemFont readedFont = new StyleItemFont();
035        V8Reader.readChar(buffer, ',');
036        readedFont.flag = V8Reader.read(Integer.class, buffer);
037        V8Reader.readChar(buffer, ',');
038        readedFont.types = V8Reader.read(IntObject.class, buffer);
039        if ((readedFont.flag & FONT_WEIGHT) == FONT_WEIGHT) {
040          V8Reader.readChar(buffer, ',');
041          readedFont.weight = V8Reader.read(Integer.class, buffer);
042        }
043        if ((readedFont.flag & FONT_UNK1) == FONT_UNK1) {
044          V8Reader.readChar(buffer, ',');
045          readedFont.unk1 = V8Reader.read(Integer.class, buffer);
046        }
047        if ((readedFont.flag & FONT_UNK2) == FONT_UNK2) {
048          V8Reader.readChar(buffer, ',');
049          readedFont.unk2 = V8Reader.read(Integer.class, buffer);
050        }
051        if ((readedFont.flag & FONT_UNK3) == FONT_UNK3) {
052          V8Reader.readChar(buffer, ',');
053          readedFont.unk3 = V8Reader.read(Integer.class, buffer);
054        }
055        if ((readedFont.flag & FONT_UNK4) == FONT_UNK4) {
056          V8Reader.readChar(buffer, ',');
057          readedFont.unk4 = V8Reader.read(Integer.class, buffer);
058        }
059        innerFont = readedFont;
060      break;
061      case WindowsFont:
062        final WindowsFont readedFont1 = new WindowsFont();
063        V8Reader.readChar(buffer, ',');
064        readedFont1.flag = V8Reader.read(Integer.class, buffer);
065        V8Reader.readChar(buffer, ',');
066        readedFont1.types = V8Reader.read(IntObject.class, buffer);
067        if ((readedFont1.flag & FONT_WEIGHT) == FONT_WEIGHT) {
068          V8Reader.readChar(buffer, ',');
069          readedFont1.weight = V8Reader.read(Integer.class, buffer);
070        }
071        if ((readedFont1.flag & FONT_UNK1) == FONT_UNK1) {
072          V8Reader.readChar(buffer, ',');
073          readedFont1.unk1 = V8Reader.read(Integer.class, buffer);
074        }
075        if ((readedFont1.flag & FONT_UNK2) == FONT_UNK2) {
076          V8Reader.readChar(buffer, ',');
077          readedFont1.unk2 = V8Reader.read(Integer.class, buffer);
078        }
079        if ((readedFont1.flag & FONT_UNK3) == FONT_UNK3) {
080          V8Reader.readChar(buffer, ',');
081          readedFont1.unk3 = V8Reader.read(Integer.class, buffer);
082        }
083        if ((readedFont1.flag & FONT_UNK4) == FONT_UNK4) {
084          V8Reader.readChar(buffer, ',');
085          readedFont1.unk4 = V8Reader.read(Integer.class, buffer);
086        }
087        if ((readedFont1.flag & FONT_UNK5) == FONT_UNK5) {
088          V8Reader.readChar(buffer, ',');
089          readedFont1.unk5 = V8Reader.read(Integer.class, buffer);
090        }
091        innerFont = readedFont1;
092      break;
093      default:
094        throw new RuntimeException("Unknown font type: " + typeFont);
095    }
096    return innerFont;
097  }
098
099  @Override
100  public void write(Object object, SerializedOutputStream buffer) {
101    throw new UnsupportedOperationException("Not implemented");
102  }
103
104}
