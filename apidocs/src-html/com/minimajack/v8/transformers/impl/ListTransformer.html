001package com.minimajack.v8.transformers.impl;
002
003import com.minimajack.v8.transformers.AbstractParametrizedTransformer;
004import com.minimajack.v8.utility.SerializedOutputStream;
005import com.minimajack.v8.utility.V8Reader;
006
007import java.lang.reflect.ParameterizedType;
008import java.lang.reflect.Type;
009import java.nio.ByteBuffer;
010import java.util.ArrayList;
011import java.util.List;
012
013public class ListTransformer
014    implements AbstractParametrizedTransformer<List<?>>
015{
016
017    @Override
018    public List<?> read( ParameterizedType type, ByteBuffer buffer )
019    {
020        Type[] types = type.getActualTypeArguments();
021        Class<?> firstClass = (Class<?>) types[0];
022        return this.read( firstClass, buffer );
023    }
024
025    @Override
026    public void write(Object object, SerializedOutputStream buffer) {
027        List<Object> list = (List<Object>)object;
028        Integer count = list.size();
029        V8Reader.write(count, buffer);
030        for( Object element : list ){
031            buffer.putComa( );
032            V8Reader.write( element, buffer );
033        }
034
035    }
036
037    public List<?> read( Class<?> clazz, ByteBuffer buffer )
038    {
039        Integer count = V8Reader.read( Integer.class, buffer );
040        List<Object> list = new ArrayList<>(count);
041
042        for ( int i = 0; i < count; i++ )
043        {
044            V8Reader.readChar( buffer, ',' );
045            list.add( V8Reader.read( clazz, buffer ) );
046
047        }
048        return list;
049    }
050}
