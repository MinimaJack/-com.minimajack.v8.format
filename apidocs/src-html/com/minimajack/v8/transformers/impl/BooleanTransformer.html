001package com.minimajack.v8.transformers.impl;
002
003import com.minimajack.v8.transformers.AbstractTransformer;
004import com.minimajack.v8.utility.SerializedOutputStream;
005import com.minimajack.v8.utility.V8Reader;
006
007import java.nio.ByteBuffer;
008
009public class BooleanTransformer
010        implements AbstractTransformer<Boolean> {
011
012    @Override
013    public Boolean read(ByteBuffer buffer) {
014        return V8Reader.read(Integer.class, buffer) == 1 ? Boolean.TRUE : Boolean.FALSE;
015    }
016
017    @Override
018    public void write(Object object, SerializedOutputStream buffer) {
019        buffer.write((Boolean)object? 0x31 : 0x30);
020    }
021
022}
