001package com.minimajack.v8.transformers.impl;
002
003import java.lang.reflect.ParameterizedType;
004import java.nio.ByteBuffer;
005
006import com.minimajack.v8.transformers.AbstractTransformer;
007import com.minimajack.v8.utility.V8Reader;
008
009public class BooleanTransformer
010    extends AbstractTransformer<Boolean>
011{
012
013    @Override
014    public Boolean read( ParameterizedType type, ByteBuffer buffer )
015    {
016        return V8Reader.read( Integer.class, buffer ) == 1 ?  Boolean.TRUE : Boolean.FALSE;
017    }
018
019}
