001package com.minimajack.v8.transformers.impl;
002
003import com.minimajack.v8.transformers.AbstractClassTransformer;
004import com.minimajack.v8.utility.SerializedOutputStream;
005import com.minimajack.v8.utility.V8Reader;
006
007import java.lang.reflect.Array;
008import java.nio.ByteBuffer;
009import java.util.ArrayList;
010import java.util.List;
011
012public class ArraysTransformer
013    extends AbstractClassTransformer<Object[]>
014{
015
016    @Override
017    public Object[] read( Class<?> clazz, ByteBuffer buffer )
018    {
019        Class<?> componentType = clazz.getComponentType();
020        List<Object> list = new ArrayList<>();
021        boolean readNext = true;
022        while ( readNext )
023        {
024            list.add( V8Reader.read( componentType, buffer ) );
025            readNext = buffer.hasRemaining() && buffer.get( buffer.position() ) == 0x2C;
026            if ( readNext )
027            {
028                V8Reader.readChar( buffer, ',' );
029            }
030        }
031        Object[] arr = (Object[]) Array.newInstance( componentType, 0 );
032        return list.toArray(arr);
033    }
034
035    @Override
036    public void write(Object object, SerializedOutputStream buffer) {
037        this.write((Object[])object, buffer);
038    }
039
040    public void write(Object[] object, SerializedOutputStream buffer ){
041
042        for (int i = 0; i <object.length; i++) {
043            V8Reader.write(object[i], null, buffer);
044            if(i != object.length-1)
045                buffer.putComa();
046        }
047    }
048
049}
