001package com.minimajack.v8.transformers.impl;
002
003import com.minimajack.v8.exeptions.NotValidUUID;
004import com.minimajack.v8.transformers.AbstractTransformer;
005import com.minimajack.v8.utility.SerializedOutputStream;
006
007import java.nio.ByteBuffer;
008import java.nio.charset.StandardCharsets;
009import java.util.Arrays;
010import java.util.UUID;
011
012/**
013 * Read UUID
014 * @author e.vanzhula
015 *
016 */
017public class UUIDTransformer
018    implements AbstractTransformer<UUID>
019{
020
021    /**
022     * Represent default UUID size
023     */
024    private static final int UUID_SIZE_IN_STRING = 36;
025    private static final byte[] ZERO_STRING = "00000000-0000-0000-0000-000000000000".getBytes(StandardCharsets.UTF_8);
026    public static final UUID ZERO_UUID = UUID.fromString("00000000-0000-0000-0000-000000000000");
027    @Override
028    public UUID read( ByteBuffer buffer )
029    {
030        byte[] guid = new byte[UUID_SIZE_IN_STRING];
031        buffer.get( guid );
032
033        if(Arrays.equals(guid, ZERO_STRING)){
034            return ZERO_UUID;
035        }
036        try
037        {
038            String guidString = new String( guid, StandardCharsets.UTF_8 );
039            return UUID.fromString( guidString );
040        }
041        catch ( Exception e )
042        {
043            throw new NotValidUUID( e );
044        }
045    }
046
047    @Override
048    public void write(Object object, SerializedOutputStream buffer) {
049        buffer.writeBytes(object.toString().getBytes(StandardCharsets.UTF_8));
050    }
051
052}
