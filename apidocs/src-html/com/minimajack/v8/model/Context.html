001package com.minimajack.v8.model;
002
003import com.minimajack.v8.format.Container;
004import com.minimajack.v8.io.reader.AbstractReader;
005
006import java.io.IOException;
007
008public class Context
009    extends AbstractContextHolder
010{
011    private String name;
012
013    private boolean inflated = false;
014
015    private Class<? extends AbstractReader> readerClass;
016
017    public Context getParent()
018    {
019        return this.getContext();
020    }
021
022    public Context createChildContext( String name )
023    {
024        Context child = new Context();
025        child.setContext( this );
026        child.setName( name );
027        return child;
028    }
029
030    public void parseContainer( final Container container )
031        throws IOException
032    {
033        AbstractReader reader;
034        try
035        {
036            reader = readerClass.newInstance();
037            reader.setContext( this );
038            reader.setContainer( container );
039            reader.read();
040        }
041        catch ( InstantiationException | IllegalAccessException | IllegalArgumentException | SecurityException e )
042        {
043            throw new RuntimeException( "Exception in reader" );
044        }
045    }
046
047    public String getName()
048    {
049        return name;
050    }
051
052    public void setName( String name )
053    {
054        this.name = name.trim();
055    }
056
057    public String getPath()
058    {
059        if ( this.getContext() != null )
060        {
061            return this.getContext().getPath() + "/" + name;
062        }
063        else
064        {
065            return name;
066        }
067    }
068
069    public boolean isInflated()
070    {
071        return inflated;
072    }
073
074    public Class<? extends AbstractReader> getReader()
075    {
076        return readerClass;
077    }
078
079    public void setReader( Class<? extends AbstractReader> reader )
080    {
081        this.readerClass = reader;
082    }
083
084    public void setInflated( boolean inflated )
085    {
086        this.inflated = inflated;
087    }
088
089}
