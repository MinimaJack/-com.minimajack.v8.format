001package com.minimajack.v8.utility;
002
003import com.minimajack.v8.transformers.AbstractClassTransformer;
004import com.minimajack.v8.transformers.AbstractParametrizedTransformer;
005import com.minimajack.v8.transformers.AbstractTransformer;
006import com.minimajack.v8.transformers.impl.*;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.lang.reflect.ParameterizedType;
011import java.lang.reflect.Type;
012import java.nio.ByteBuffer;
013import java.util.*;
014
015public class V8Reader {
016
017    final static Logger logger = LoggerFactory.getLogger( V8Reader.class );
018
019    private static final byte SPACE = 0x20;
020
021    private static final byte CR = 0x0D;
022
023    private static final byte LF = 0x0A;
024
025    private static final Map<Class<?>, AbstractTransformer<?>> TRANSFORMERS = new HashMap<>();
026
027    private static final Map<Class<?>, AbstractParametrizedTransformer<?>> PARAMETERIZED_TRANSFORMERS = new HashMap<>();
028
029    private static final Map<Class<?>, AbstractClassTransformer<?>> CLASS_TRANSFORMERS = new HashMap<>();
030
031    public static <T> T read(Class<? extends T> clazz, ByteBuffer buffer) {
032        return read(clazz, null, buffer);
033    }
034
035    @SuppressWarnings("unchecked")
036    public static <T> T read(Class<T> clazz, Type type, ByteBuffer buffer) {
037        logger.debug("Read class {}", clazz.getName());
038        T res = null;
039        if (TRANSFORMERS.containsKey(clazz)) {
040            if (type instanceof ParameterizedType) {
041               throw new RuntimeException("Must not be here");
042            } else {
043                res = (T) TRANSFORMERS.get(clazz).read(buffer);
044            }
045        } else if(PARAMETERIZED_TRANSFORMERS.containsKey(clazz)){
046            if (PARAMETERIZED_TRANSFORMERS.containsKey(clazz)) {
047                if (type instanceof ParameterizedType) {
048                    res = (T) PARAMETERIZED_TRANSFORMERS.get(clazz).read((ParameterizedType) type, buffer);
049                } else {
050                    throw new RuntimeException("Must not be here");
051                }
052            }
053        }
054        else if (clazz.isEnum()) {
055            res = (T) CLASS_TRANSFORMERS.get(Enum.class).read(clazz, buffer);
056        } else if (clazz.isArray()) {
057            res = (T) CLASS_TRANSFORMERS.get(Arrays.class).read(clazz, buffer);
058        } else {
059            res = (T) CLASS_TRANSFORMERS.get(Object.class).read(clazz, buffer);
060        }
061        logger.debug("End read class {}", clazz.getName());
062        return res;
063
064    }
065
066
067    public static void write(Object object, SerializedOutputStream buffer) {
068       write(object, null, buffer);
069    }
070    public static void write(Object object, Type type, SerializedOutputStream buffer) {
071        Class<?> clazz = object.getClass();
072        if (TRANSFORMERS.containsKey(clazz)) {
073            TRANSFORMERS.get(clazz).write(object, buffer);
074        }  else if(PARAMETERIZED_TRANSFORMERS.containsKey(clazz)){
075             PARAMETERIZED_TRANSFORMERS.get(clazz).write(object, buffer);
076        }else if (clazz.isEnum()) {
077            CLASS_TRANSFORMERS.get(Enum.class).write(object, buffer);
078        } else if (clazz.isArray()) {
079            CLASS_TRANSFORMERS.get(Arrays.class).write(object, buffer);
080        } else {
081            CLASS_TRANSFORMERS.get(Object.class).write(object, buffer);
082        }
083    }
084
085    public static void init() {
086        TRANSFORMERS.put(String.class, new StringTransformer());
087        TRANSFORMERS.put(Integer.class, new IntegerTransformer());
088        TRANSFORMERS.put(UUID.class, new UUIDTransformer());
089        TRANSFORMERS.put(Boolean.class, new BooleanTransformer());
090        TRANSFORMERS.put(AnyData.class, new AnyDataTransformer());
091
092        PARAMETERIZED_TRANSFORMERS.put(List.class, new ListTransformer());
093        PARAMETERIZED_TRANSFORMERS.put(LinkedList.class, new ListTransformer());
094
095        PARAMETERIZED_TRANSFORMERS.put(Map.class, new MapTransformer());
096        PARAMETERIZED_TRANSFORMERS.put(LinkedHashMap.class, new MapTransformer());
097
098        CLASS_TRANSFORMERS.put(Object.class, new ObjectTransformer());
099        CLASS_TRANSFORMERS.put(Enum.class, new EnumsTransformer());
100        CLASS_TRANSFORMERS.put(Arrays.class, new ArraysTransformer());
101    }
102
103    public static void registerTransformer(Class<?> clazz, AbstractTransformer<?> transformer) {
104        TRANSFORMERS.put(clazz, transformer);
105    }
106
107    public static void registerParametrizedTransformer(Class<?> clazz, AbstractParametrizedTransformer<?> transformer) {
108        PARAMETERIZED_TRANSFORMERS.put(clazz, transformer);
109    }
110
111    public static void readChar(ByteBuffer buffer, char ch) {
112        while (buffer.hasRemaining()) {
113            int value = buffer.get() & 0xFF;
114
115            if ((char) value == ch) {
116                break;
117            } else if (value != SPACE && value != CR && value != LF) {
118                throw new RuntimeException("Bad chars in buffer [" + String.valueOf(Character.toChars(value)) + "] must be: [" + ch + "]" + " position: "
119                        + (buffer.position() - 1));
120            }
121        }
122    }
123
124}
