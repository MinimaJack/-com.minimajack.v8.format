001package com.minimajack.v8.utility;
002
003import java.lang.reflect.ParameterizedType;
004import java.lang.reflect.Type;
005import java.nio.ByteBuffer;
006import java.util.Arrays;
007import java.util.HashMap;
008import java.util.List;
009import java.util.Map;
010import java.util.UUID;
011
012import com.minimajack.v8.transformers.AbstractArraysTransformer;
013import com.minimajack.v8.transformers.AbstractClassTransformer;
014import com.minimajack.v8.transformers.AbstractEnumTransformer;
015import com.minimajack.v8.transformers.AbstractTransformer;
016import com.minimajack.v8.transformers.impl.AnyDataTransformer;
017import com.minimajack.v8.transformers.impl.ArraysTransformer;
018import com.minimajack.v8.transformers.impl.BooleanTransformer;
019import com.minimajack.v8.transformers.impl.ClassTransformer;
020import com.minimajack.v8.transformers.impl.EnumsTransformer;
021import com.minimajack.v8.transformers.impl.IntegerTransformer;
022import com.minimajack.v8.transformers.impl.ListTransformer;
023import com.minimajack.v8.transformers.impl.MapTransformer;
024import com.minimajack.v8.transformers.impl.StringTransformer;
025import com.minimajack.v8.transformers.impl.UUIDTransformer;
026
027public class V8Reader
028{
029
030    private static final byte SPACE = 0x20;
031
032    private static final byte CR = 0x0D;
033
034    private static final byte LF = 0x0A;
035
036    private static final Map<Class<?>, AbstractTransformer<?>> TRANSFORMERS = new HashMap<>();
037
038    private static final Map<Class<?>, AbstractEnumTransformer<?>> ENUM_TRANSFORMERS = new HashMap<>();
039
040    private static final Map<Class<?>, AbstractArraysTransformer<?>> ARRAYS_TRANSFORMERS = new HashMap<>();
041
042    private static final Map<Class<?>, AbstractClassTransformer<?>> CLASS_TRANSFORMERS = new HashMap<>();
043
044    public static <T> T read( Class<? extends T> clazz, ByteBuffer buffer )
045    {
046        return read( clazz, null, buffer );
047    }
048
049    @SuppressWarnings("unchecked")
050    public static <T> T read( Class<T> clazz, Type type, ByteBuffer buffer )
051    {
052        if ( TRANSFORMERS.containsKey( clazz ) )
053        {
054            if ( type instanceof ParameterizedType )
055            {
056                return (T) TRANSFORMERS.get( clazz ).read( (ParameterizedType) type, buffer );
057            }
058            else
059            {
060                return (T) TRANSFORMERS.get( clazz ).read( null, buffer );
061            }
062        }
063        if ( clazz.isEnum() )
064        {
065            return (T) ENUM_TRANSFORMERS.get( Enum.class ).read( clazz, buffer );
066        }
067        if ( clazz.isArray() )
068        {
069            return (T) ARRAYS_TRANSFORMERS.get( Arrays.class ).read( clazz, buffer );
070        }
071        return (T) CLASS_TRANSFORMERS.get( Object.class ).read( clazz, buffer );
072    }
073
074    public static final void init()
075    {
076        TRANSFORMERS.put( String.class, new StringTransformer() );
077        TRANSFORMERS.put( Integer.class, new IntegerTransformer() );
078        TRANSFORMERS.put( List.class, new ListTransformer() );
079        TRANSFORMERS.put( Map.class, new MapTransformer() );
080        TRANSFORMERS.put( UUID.class, new UUIDTransformer() );
081        TRANSFORMERS.put( Boolean.class, new BooleanTransformer() );
082        TRANSFORMERS.put( AnyData.class, new AnyDataTransformer() );
083
084        CLASS_TRANSFORMERS.put( Object.class, new ClassTransformer() );
085        ENUM_TRANSFORMERS.put( Enum.class, new EnumsTransformer() );
086        ARRAYS_TRANSFORMERS.put( Arrays.class, new ArraysTransformer() );
087    }
088
089    public static void registerTransformer( Class<?> clazz, AbstractTransformer<?> transformer )
090    {
091        TRANSFORMERS.put( clazz, transformer );
092    }
093
094    public static void readChar( ByteBuffer buffer, char ch )
095    {
096        while ( buffer.hasRemaining() )
097        {
098            int value = buffer.get() & 0xFF;
099
100            if ( (char) value == ch )
101            {
102                break;
103            }
104            else if ( value != SPACE && value != CR && value != LF )
105            {
106                throw new RuntimeException( "Bad chars in buffer " + value + " [" + ch + "]" + " position: "
107                    + ( buffer.position() - 1 ) );
108            }
109        }
110    }
111
112}
