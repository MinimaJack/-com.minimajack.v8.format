001package com.minimajack.v8.utils;
002
003import java.io.IOException;
004import java.io.InputStream;
005
006public class V8StreamUtils
007{
008    private static final char[] HEX_ARRAY = "0123456789ABCDEF".toCharArray();
009
010    public static boolean isContainerStream( InputStream is )
011        throws IOException
012    {
013        return is.read() == 0xFF && is.read() == 0xFF && is.read() == 0xFF && is.read() == 0x7F;
014    }
015
016    public static String bytesToHex( byte[] bytes )
017    {
018        char[] hexChars = new char[bytes.length * 2];
019        for ( int j = 0; j < bytes.length; j++ )
020        {
021            int v = bytes[j] & 0xFF;
022            hexChars[j * 2] = HEX_ARRAY[v >>> 4];
023            hexChars[j * 2 + 1] = HEX_ARRAY[v & 0x0F];
024        }
025        return new String( hexChars );
026    }
027}
