001package com.minimajack.v8.utils;
002
003import java.util.Date;
004
005public class DateUtils
006{
007    private static final long MILLISEC_TO_UNIXTIME_X10 = 621356075920000L;
008
009    public static long tov8Time( Date date )
010    {
011        return MILLISEC_TO_UNIXTIME_X10 + date.getTime() * 10;
012    }
013
014    public static Date fromv8Time( long v8date )
015    {
016        if(v8date == 0){
017            return null;
018        }
019        return new Date( ( v8date - MILLISEC_TO_UNIXTIME_X10 ) / 10 );
020    }
021}
