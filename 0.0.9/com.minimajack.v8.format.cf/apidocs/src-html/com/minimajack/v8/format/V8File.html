001package com.minimajack.v8.format;
002
003import java.io.DataOutput;
004import java.io.IOException;
005import java.nio.ByteBuffer;
006
007import com.minimajack.v8.model.BufferedObject;
008
009public class V8File
010    extends BufferedObject
011{
012
013    public static final int FILE_DESCRIPTION_SIZE = 12;
014
015    private int headerAddress = BlockHeader.V8_ENDBLOCK;
016
017    private int bodyAddress = BlockHeader.V8_ENDBLOCK;
018
019    private V8FileAttribute attributes;
020
021    private V8FileBody body;
022
023    public V8File()
024    {
025
026    }
027
028    public V8File( ByteBuffer buffer )
029    {
030        super( buffer );
031    }
032
033    public void read()
034        throws IOException
035    {
036        setStartPosition();
037        ByteBuffer buffer = getBuffer();
038        this.headerAddress = buffer.getInt() & 0xFFFFFFFF;
039        this.bodyAddress = buffer.getInt() & 0xFFFFFFFF;
040        int reserved = buffer.getInt() & 0xFFFFFFFF;
041        if ( reserved != Integer.MAX_VALUE )
042        {
043            throw new RuntimeException( "Bad magic number" );
044        }
045    }
046
047    @Override
048    public void write( DataOutput buffer )
049        throws IOException
050    {
051        buffer.writeInt( this.headerAddress );
052        buffer.writeInt( this.bodyAddress );
053        buffer.writeInt( Integer.MAX_VALUE );
054    }
055
056    public boolean isContainer()
057        throws IOException
058    {
059        return this.body != null && this.body.isContainer();
060    }
061
062    public void readHeader( ByteBuffer dataBufer )
063        throws IOException
064    {
065        if ( this.headerAddress != BlockHeader.V8_ENDBLOCK )
066        {
067            this.attributes = new V8FileAttribute( dataBufer, this.headerAddress );
068            this.attributes.setContext( getContext() );
069            this.attributes.read();
070        }
071    }
072
073    public void readBody( ByteBuffer dataBufer )
074        throws IOException
075    {
076        if ( this.headerAddress != BlockHeader.V8_ENDBLOCK && this.bodyAddress != BlockHeader.V8_ENDBLOCK )
077        {
078            this.body = new V8FileBody( dataBufer, this.bodyAddress );
079            this.body.setContext( getContext() );
080            this.body.read();
081        }
082    }
083
084    public V8FileAttribute getAttributes()
085    {
086        return attributes;
087    }
088
089    public void setAttributes( V8FileAttribute attributes )
090    {
091        this.attributes = attributes;
092        this.headerAddress = attributes.getPosition();
093    }
094
095    public V8FileBody getBody()
096    {
097        return body;
098    }
099
100    public void setBody( V8FileBody body )
101    {
102        this.body = body;
103        this.bodyAddress = body.getPosition();
104    }
105
106    public int getHeaderAddress()
107    {
108        return headerAddress;
109    }
110
111    public void setHeaderAddress( int headerAddress )
112    {
113        this.headerAddress = headerAddress;
114    }
115
116    public int getBodyAddress()
117    {
118        return bodyAddress;
119    }
120
121    public void setBodyAddress( int bodyAddress )
122    {
123        this.bodyAddress = bodyAddress;
124    }
125
126    @Override
127    public int getSize()
128        throws IOException
129    {
130        return FILE_DESCRIPTION_SIZE;
131    }
132
133}
